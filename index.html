-- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô GiveXP ‡πÅ‡∏•‡∏∞ Trion Recharge
local function GiveXP(player, amount)
    local stats = player:FindFirstChild("PlayerStats")
    if not stats then return end
    stats.XP.Value = stats.XP.Value + amount
    local req = stats.Level.Value * 100 
    while stats.XP.Value >= req do
        stats.Level.Value = stats.Level.Value + 1
        stats.XP.Value = stats.XP.Value - req
        stats.StatPoints.Value = stats.StatPoints.Value + 3
        req = stats.Level.Value * 100 
    end
end
-- Trion Recharge Loop (‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏ó‡∏∏‡∏Å 1 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ)
game:GetService("RunService").Heartbeat:Connect(function() 
    for _, player in ipairs(game.Players:GetPlayers()) do
        local stats = player:FindFirstChild("PlayerStats")
        if stats then
            local max = stats.TrionCapacity.Value
            local current = stats.CurrentTrion
            current.Value = math.min(current.Value + 5, max)
        end
    end
end)

-- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö Server ‡πÉ‡∏ä‡πâ‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏™‡πÄ‡∏ï‡∏ï‡∏±‡∏™‡∏à‡∏£‡∏¥‡∏á
Config.OnAddStatFired = function(player, statName)
    local stats = player:FindFirstChild("PlayerStats")
    if stats and stats.StatPoints.Value >= 1 then
        local stat = stats:FindFirstChild(statName)
        if stat and stat:IsA("IntValue") then
            stats.StatPoints.Value = stats.StatPoints.Value - 1
            stat.Value = stat.Value + 1
            return true
        end
    end
    return false
end
return Config
